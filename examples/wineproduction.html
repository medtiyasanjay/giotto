The total wine production in 2011 was 267,413 million of ectoliters, source OIV.

Html:

<div data-options="gexamples.piecharts" style="width: 100%;" giotto-pie></div>

gexamples.piecharts = {
    height: '80%',

    data: [{pie: oivdata}],

    // Callback for angular directive
    angular: function (chart, opts) {

        function update (form) {
            angular.forEach(form, function (value, name) {
                if (name === 'type')
                    opts.type = value;
                else
                    chart.setSerieOption('pie', name, value);
            });
            chart.resume();
        }

        opts.scope.$on('formReady', function (e, form) {
            update(form);
        });
        opts.scope.$on('formFieldChange', function (e, form) {
            update(form);
        });
    }
};
