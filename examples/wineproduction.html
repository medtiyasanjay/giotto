2011 World Wine Production

The total wine production in 2011 was 267,413 million of ectoliters, source OIV.

Html:

<div data-options="gexamples.piecharts" style="width: 100%;" giotto-pie></div>

gexamples.piecharts = {
    height: '80%',
    margin: 80,

    tooltip: {
      show: true
    },

    pie: {
        show: true,
        formatY: function (d) {
            return d3.format(',')(d) + 'M ectoliters';
        },
        active: {
            outerRadius: '105%'
        },
        labels: {
            position: 'outside'
        },
        formatPercent: ',.1%'
    },

    data: [{label: 'Wine Production 2013', data: oivdata}],

    // Callback for angular directive
    angular: function (chart, opts) {

        function update (form) {
            angular.forEach(form, function (value, name) {
                if (name === 'type')
                    opts.type = value;
                else
                    chart.setSerieOption('pie', name, value);
            });
            chart.resume();
        }

        chart.scope().$on('formReady', function (e, form) {
            update(form);
        });
        chart.scope().$on('formFieldChange', function (e, form) {
            update(form);
        });
    }
};
